setup() { // 1. 一開始給空物件，等一下從 JSON 讀取 const etfDb = ref({}); // 使用者狀態 const selectedEtfs = ref(["0056", "00878", "00919", "00929"]); const etfShares = ref({ "0056": 10000, "00878": 15000, "00919": 12000, "00929": 12000, "00940": 0, "00918": 0, "00713": 0 }); let chartInstance = null; // 2. 建立網頁載入時讀取 JSON 的邏輯 onMounted(async () => { try { // 讀取同資料夾底下的 json 檔案 const response = await fetch('./etf_data.json'); etfDb.value = await response.json(); // 資料讀取成功後，再初始化圖表 initChart(); } catch (error) { console.error("讀取 ETF 資料失敗:", error); } }); // 以下計算邏輯因為 etfDb 變成了 Ref，所以裡面用到 etfDb 的地方都要加上 .value const totalMarketValue = computed(() => { if (!etfDb.value["0056"]) return 0; // 確保資料已載入 return selectedEtfs.value.reduce((sum, code) => { const shares = etfShares.value[code] || 0; return sum + (shares * etfDb.value[code].price); }, 0); }); const totalAnnualDividend = computed(() => { if (!etfDb.value["0056"]) return 0; return selectedEtfs.value.reduce((sum, code) => { const shares = etfShares.value[code] || 0; return sum + (shares * etfDb.value[code].dividend); }, 0); }); const totalMonthlyDividend = computed(() => Math.round(totalAnnualDividend.value / 12)); const averageYield = computed(() => { if (totalMarketValue.value === 0) return "0.00"; return ((totalAnnualDividend.value / totalMarketValue.value) * 100).toFixed(2); }); const monthlyData = computed(() => { let months = new Array(12).fill(0); if (!etfDb.value["0056"]) return months; selectedEtfs.value.forEach(code => { const etf = etfDb.value[code]; const shares = etfShares.value[code] || 0; etf.months.forEach(m => { months[m - 1] += Math.round(shares * etf.perTime); }); }); return months; }); const updateChart = () => { if (!chartInstance) return; chartInstance.data.datasets[0].data = monthlyData.value; chartInstance.update(); }; // 將圖表初始化獨立成一個 function const initChart = () => { const ctx = document.getElementById('cashFlowChart').getContext('2d'); chartInstance = new Chart(ctx, { type: 'bar', data: { labels: ['1月', '2月', '3月', '4月', '5月', '6月', '7月', '8月', '9月', '10月', '11月', '12月'], datasets: [{ label: '預估配息金額 (元)', data: monthlyData.value, backgroundColor: '#f97316', borderRadius: 6 }] }, options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false } }, scales: { y: { grid: { color: '#374151' }, ticks: { color: '#9ca3af' } }, x: { grid: { display: false }, ticks: { color: '#9ca3af' } } } } }); }; return { etfDb, selectedEtfs, etfShares, totalMarketValue, totalAnnualDividend, totalMonthlyDividend, averageYield, updateChart }; }
